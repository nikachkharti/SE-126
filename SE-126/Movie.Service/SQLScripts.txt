CREATE PROCEDURE sp_getAllCountries
AS
BEGIN
	SELECT
	  [CountryID],
      [Country]
	FROM [Movies].[dbo].[Country]
END



CREATE PROCEDURE sp_addCountry
@countryName NVARCHAR(255)
AS
BEGIN
	INSERT INTO Country (Country)
	VALUES(@countryName)
END


CREATE PROCEDURE [dbo].[sp_getSingleCountry]
@id INT
AS
BEGIN
	SELECT
	  [CountryID],
      [Country]
	FROM [Movies].[dbo].[Country]
	WHERE CountryID = @id
END

CREATE PROCEDURE sp_getAllStudios
AS
BEGIN
	SELECT
		StudioID,
		Studio
	FROM Studio
END


CREATE PROCEDURE sp_updateCountry
@countryId INT,
@countryName NVARCHAR(255)
AS
BEGIN
	UPDATE Country
	SET Country = @countryName
	WHERE CountryID = @countryId
END

CREATE PROCEDURE [dbo].[sp_deleteCountry]
@countryId INT
AS
BEGIN
	DELETE FROM
	Country
	WHERE CountryID = @countryId
END



CREATE PROCEDURE sp_getSingleDirector
@id INT
AS
BEGIN
SELECT
	   [DirectorID]
      ,[FirstName]
      ,[FamilyName]
      ,[FullName]
      ,[DoB]
      ,[DoD]
      ,[Gender]
  FROM [Movies].[dbo].[Director]
  WHERE DirectorID = @id
END



CREATE PROCEDURE sp_getAllDirectors
AS
BEGIN
SELECT
	   [DirectorID]
      ,[FirstName]
      ,[FamilyName]
      ,[FullName]
      ,[DoB]
      ,[DoD]
      ,[Gender]
  FROM [Movies].[dbo].[Director]
END


CREATE PROCEDURE sp_addDirector
@firstName NVARCHAR(100),
@familyName NVARCHAR(100),
@dob DATETIME,
@dod DATETIME,
@gender NVARCHAR(20)
AS
BEGIN
	INSERT INTO Director(FirstName,FamilyName,DoB,DoD,Gender)
	VALUES(@firstName,@familyName,@dob,@dod,@gender)
END


CREATE PROCEDURE sp_updateDirector
@id INT,
@firstName NVARCHAR(100),
@familyName NVARCHAR(100),
@dob DATETIME,
@dod DATETIME,
@gender NVARCHAR(20)
AS
BEGIN
	UPDATE Director
	SET
		FirstName = @firstName,
		FamilyName = @familyName,
		DoB = @dob,
		DoD = @dod,
		Gender = @gender
	WHERE DirectorID = @id
END


CREATE PROCEDURE sp_deleteDirector
@id INT
AS
BEGIN
	DELETE Director
	WHERE DirectorID = @id
END
