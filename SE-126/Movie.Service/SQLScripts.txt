//Country

CREATE PROCEDURE sp_getAllCountries
AS
BEGIN
	SELECT
	  [CountryID],
      [Country]
	FROM [Movies].[dbo].[Country]
END



CREATE PROCEDURE sp_addCountry
@param0 NVARCHAR(255)
AS
BEGIN
	INSERT INTO Country (Country)
	VALUES(@param0)
END


CREATE PROCEDURE [dbo].[sp_getSingleCountry]
@param0 INT
AS
BEGIN
	SELECT
	  [CountryID],
      [Country]
	FROM [Movies].[dbo].[Country]
	WHERE CountryID = @param0
END

CREATE PROCEDURE sp_getAllStudios
AS
BEGIN
	SELECT
		StudioID,
		Studio
	FROM Studio
END


CREATE PROCEDURE sp_updateCountry
@param0 INT,
@param1 NVARCHAR(255)
AS
BEGIN
	UPDATE Country
	SET Country = @param1
	WHERE CountryID = @param0
END

CREATE PROCEDURE [dbo].[sp_deleteCountry]
@param0 INT
AS
BEGIN
	DELETE FROM
	Country
	WHERE CountryID = @param0
END

//Director

CREATE PROCEDURE sp_getSingleDirector
@param0 INT
AS
BEGIN
SELECT
	   [DirectorID]
      ,[FirstName]
      ,[FamilyName]
      ,[FullName]
      ,[DoB]
      ,[DoD]
      ,[Gender]
  FROM [Movies].[dbo].[Director]
  WHERE DirectorID = @param0
END



CREATE PROCEDURE sp_getAllDirectors
AS
BEGIN
SELECT
	   [DirectorID]
      ,[FirstName]
      ,[FamilyName]
      ,[FullName]
      ,[DoB]
      ,[DoD]
      ,[Gender]
  FROM [Movies].[dbo].[Director]
END


CREATE PROCEDURE sp_addDirector
@param0 NVARCHAR(100),
@param1 NVARCHAR(100),
@param2 DATETIME,
@param3 DATETIME,
@param4 NVARCHAR(20)
AS
BEGIN
	INSERT INTO Director(FirstName,FamilyName,DoB,DoD,Gender)
	VALUES(@param0,@param1,@param2,@param3,@param4)
END


CREATE PROCEDURE sp_updateDirector
@param0 INT,
@param1 NVARCHAR(100),
@param2 NVARCHAR(100),
@param3 DATETIME,
@param4 DATETIME,
@param5 NVARCHAR(20)
AS
BEGIN
	UPDATE Director
	SET
		FirstName = @param1,
		FamilyName = @param2,
		DoB = @param3,
		DoD = @param4,
		Gender = @param5
	WHERE DirectorID = @param0
END


CREATE PROCEDURE sp_deleteDirector
@param0 INT
AS
BEGIN
	DELETE Director
	WHERE DirectorID = @param0
END
